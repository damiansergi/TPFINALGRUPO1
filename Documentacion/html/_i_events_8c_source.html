38 unsigned char getInputEvent (void){
39 
40  unsigned char salida;
41  if (*bufferPointer == VACIO){ // Si no hay nada escrito simplemente devuelvo eso
42  salida = VACIO;
43  } else {
44  if (bufferPointer <= inputBuffer + MAXIMOEVENTOSBUFFER && bufferPointer > inputBuffer) { //Si todavia no llegue al final
45  salida = *bufferPointer;
46  *bufferPointer = VACIO;
47  bufferPointer--;
48  } else { //Una vez llego al "final" del buffer, empiezo de vuelta y sigo
49  salida = *bufferPointer;
50  *bufferPointer = VACIO;
51  bufferPointer = inputBuffer + MAXIMOEVENTOSBUFFER;
52  }
53  }
54  return salida;
55 }
